Usage:  docker buildx build [OPTIONS] PATH | URL | -

            Start a build

            Aliases:
            docker build, docker builder build, docker image build, docker buildx b

            Options:
                --add-host strings              Добавьте пользовательское сопоставление хоста с IP-адресом
                                                (format: "host:ip")
                --allow strings                 Разрешить дополнительные привилегии
                                                (e.g., "network.host",
                                                "security.insecure")
                --annotation stringArray        Добавьте аннотацию к изображению
                --attest stringArray            Параметры аттестации (format:
                                                "type=sbom,generator=image")
                --build-arg stringArray         Установите переменные времени сборки
                --build-context stringArray     Дополнительные контексты сборки (напр.,
                                                name=path)
                --builder string                Переопределить настроенный конструктор
                                                пример (default "default")
                --cache-from stringArray        Внешние источники кэша (напр.,
                                                "user/app:cache",
                                                "type=local,src=path/to/dir")
                --cache-to stringArray          Пункты назначения экспорта кэша (напр.,
                                                "user/app:cache",
                                                "type=local,dest=path/to/dir")
                --call string                   Установите метод для оценки сборки
                                                ("check", "outline", "targets")
                                                (default "build")
                --cgroup-parent string          Установите родительскую группу для "RUN"
                                                инструкции во время сборки
                --check                         Сокращение для "--call=check" (default )
            -D, --debug                         Включить ведение журнала отладки
            -f, --file string                   Имя файла Dockerfile (по умолчанию:
                                                "PATH/Dockerfile")
                --iidfile string                Запишите идентификатор изображения в файл
                --label stringArray             Set metadata for an image
                --load                          Shorthand for "--output=type=docker"
                --metadata-file string          Write build result metadata to a file
                --network string                Set the networking mode for the
                                                "RUN" instructions during build
                                                (default "default")
                --no-cache                      Do not use cache when building the image
                --no-cache-filter stringArray   Do not cache specified stages
            -o, --output stringArray            Output destination (format:
                                                "type=local,dest=path")
                --platform stringArray          Set target platform for build
                --progress string               Set type of progress output
                                                ("auto", "quiet", "plain", "tty",
                                                "rawjson"). Use plain to show
                                                container output (default "auto")
                --provenance string             Сокращение для "--attest=type=provenance"
                --pull                          Always attempt to pull all
                                                referenced images
                --push                          Сокращение для "--output=type=registry"
            -q, --quiet                         Suppress the build output and print
                                                image ID on success
                --sbom string                   Shorthand for "--attest=type=sbom"
                --secret stringArray            Secret to expose to the build
                                                (format:
                                                "id=mysecret[,src=/local/secret]")
                --shm-size bytes                Shared memory size for build containers
                --ssh stringArray               SSH agent socket or keys to expose
                                                to the build (format:
                                                "default|[=|[,]]")
            -t, --tag stringArray               Name and optionally a tag (format:
                                                "name:tag")
            docker build -t my_python_app .
                                                docker build -t sasha-backend:0.0.1 .
                --target string                 Set the target build stage to build
                --ulimit ulimit                 Ulimit options (default [])

            Experimental commands and flags are hidden. Set BUILDX_EXPERIMENTAL=1 to show them.
