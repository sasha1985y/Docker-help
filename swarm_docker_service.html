Usage:  docker service COMMAND

            Управление сервисами Swarm

            Commands:
            create                Создайте новый сервис
            docker service create --name my_service --network my_overlay_network nginx
            docker service create --name my_web --replicas 3 -p 8080:80 nginx
            docker service create --name webserver --network my_overlay_network -p 8080:80 nginx
            docker service create --name database --network my_overlay_network -e POSTGRES_PASSWORD=mysecretpassword postgres
            inspect               Отображение подробной информации об одной или нескольких службах
            logs                  Извлекать журналы службы или задачи
            ls                    Список услуг
            ps                    Перечислите задачи одной или нескольких служб
            rm                    Удалите одну или несколько служб
            rollback              Отменить изменения в конфигурации службы
            scale                 Масштабирование одной или нескольких реплицируемых служб
            docker service scale my_web=5
            update                Обновить службу
            docker service update --image nginx:latest my_web
            docker service update --image nginx:latest --update-order start-first my_web
