Usage: docker run [OPTIONS] IMAGE [COMMAND] [ARG…]

Create and run a new container from an image

Aliases: docker container run, docker run

                    Опции:
                --add-host list                    Добавить пользовательское картирование хоста в IP(host:ip)
                --annotation map                   Добавить аннотацию в контейнер (перенесено в OCI Runtime)(default map[])
            -a, --attach list                      Прикрепить к STDIN, STDOUT or STDERR
                --blkio-weight uint16              Блок IO (относительный вес), между 10 и 1000 или 0, чтобы отключить
                                                    (по умолчанию 0)
                --blkio-weight-device list         Блок IO Вес (относительный вес устройства) (по умолчанию [])
                --cap-add list                     Добавить возможности Linux
                --cap-drop list                    Drop Linux возможностей
                --cgroup-parent string             Необязательный родительский CGROUP для контейнера
                --cgroupns string                  Необязательный родительский CGROUP для контейнера (host|private)
                                                    'host':    Запустите контейнер в CGROUP хоста Docker
                                                    пространство имен
                                                    'private': Запустите контейнер в собственной частной CGROUP
                                                    пространство имен
                                                    '':       Используйте пространство имен cgroup в соответствии с настройками
                                                                параметр default-cgroupns-mode для демона (по умолчанию)
                --cidfile string                   Запишите идентификатор контейнера в файл
                --cpu-period int                   Ограничьте период использования CPU CFS (полностью справедливый планировщик)
                --cpu-quota int                    Ограничьте квоту CPU CFS (полностью справедливый планировщик)
                --cpu-rt-period int                Ограничение периода реального времени процессора в микросекундах
                --cpu-rt-runtime int               Ограничьте время работы процессора в реальном времени микросекундами
            -c, --cpu-shares int                   Доли процессора (относительный вес)
                --cpus decimal                     Количество процессоров
                --cpuset-cpus string               Процессоры, в которых разрешено выполнение (0-3, 0,1)
                --cpuset-mems string               MEMS, в которых можно разрешить выполнение (0-3, 0,1)
            -d, --detach                           Запустите контейнер в фоновом режиме и выведите идентификатор контейнера
                --detach-keys string               Переопределение последовательности клавиш для отсоединения контейнера
                --device list                      Добавьте хост-устройство в контейнер
                --device-cgroup-rule list          Добавьте правило в список разрешенных устройств cgroup
                --device-read-bps list             Ограничение скорости чтения (байт в секунду) с устройства (по умолчанию [])
                --device-read-iops list            Предельная скорость считывания (IO в секунду) с устройства (по умолчанию [])
                --device-write-bps list            Ограничить скорость записи (байт в секунду) на устройство (по умолчанию [])
                --device-write-iops list           Ограничить скорость записи (IO в секунду) на устройство (по умолчанию [])
                --disable-content-trust            Пропустить проверку изображения (по умолчанию true)
                --dns list                         Установка пользовательских DNS-серверов
                --dns-option list                  Установите параметры DNS
                --dns-search list                  Настройка пользовательских доменов поиска DNS
                --domainname string                Доменное имя контейнера NIS
                --entrypoint string                Перезапишите точку входа изображения по умолчанию
            -e, --env list                         Установите переменные среды
                --env-file list                    Чтение в файле переменных окружения
                --expose list                      Укажите порт или диапазон портов
                --gpus gpu-request                 Графические устройства для добавления в контейнер
                                                    ("все" для передачи всех графических процессоров)
                --group-add list                   Добавьте дополнительные группы для присоединения
                --health-cmd string                Команда, которую нужно запустить для проверки работоспособности
                --health-interval duration         Время между выполнением проверки (мс|с|м|ч) (по умолчанию 0 секунд)
                --health-retries int               Последовательные сбои, необходимые для сообщения о неработоспособности
                --health-start-interval duration   Время между выполнением проверки в течение начального периода
                                                    (ms|s|m|h) (default 0s)
                --health-start-period duration     Начальный период инициализации контейнера до
                                                    запуск работоспособности - обратный отсчет повторных попыток (мс|с|м|ч)
                                                    (по умолчанию 0 секунд)
                --health-timeout duration          Максимальное время выполнения одной проверки (мс|с|м|ч) (по умолчанию 0 секунд)
                --help                             Использование печати
            -h, --hostname string                  Имя хоста контейнера
                --init                             Запустите init внутри контейнера, который пересылает сигналы и
                                                    пожинает плоды процессов
            -i, --interactive                      Держите стандартный ввод открытым, даже если он не подключен
                --ip string                        IPv4 address (e.g., 172.30.100.104)
                --ip6 string                       IPv6 address (e.g., 2001:db8::33)
                --ipc string                       Используемый режим IPC
                --isolation string                 Технология изоляции контейнеров
                --kernel-memory bytes              Ограничение памяти ядра
            -l, --label list                       Установка метаданных в контейнер
                --label-file list                  Чтение в файле меток, разделенных строками
                --link list                        Добавить ссылку на другой контейнер
                --link-local-ip list               Контейнер IPv4/IPv6 link-локальные адреса
                --log-driver string                Драйвер ведения журнала для контейнера
                --log-opt list                     Параметры драйвера журнала
                --mac-address string               MAC-адрес контейнера (e.g., 92:d0:c6:0a:29:33)
            -m, --memory bytes                     Ограничение памяти
                --memory-reservation bytes         Мягкий предел памяти
                --memory-swap bytes                Предел подкачки, равный объему памяти плюс подкачка: '-1' для включения
                                                    неограниченный своп
                --memory-swappiness int            Настройка скорости замены памяти контейнера (от 0 до 100) (по умолчанию -1)
                --mount mount                      Прикрепите монтируемую файловую систему к контейнеру
                --name string                      Присвойте имя контейнеру
                --network network                  Подключение контейнера к сети
                --network-alias list               Добавьте сетевой псевдоним для контейнера
                --no-healthcheck                   Отключите любую проверку работоспособности, указанную в контейнере
                --oom-kill-disable                 Отключить OOM-убийцу
                --oom-score-adj int                Настройка параметров OOM хоста (от -1000 до 1000)
                --pid string                       Используемое пространство имен PID
                --pids-limit int                   Настройка лимита pid-кодов контейнера 
                                                    (установите значение -1 для неограниченного количества)
                --platform string                  Установите платформу, если сервер поддерживает работу с несколькими платформами
                --privileged                       Предоставьте расширенные привилегии этому контейнеру
            -p, --publish list                     Опубликуйте порт(ы) контейнера на хосте
            -P, --publish-all                      Опубликуйте все открытые порты на случайных портах
                --pull string                      Извлеките изображение перед запуском ("всегда", "отсутствует", "никогда")
                                                    (по умолчанию "отсутствует")
            -q, --quiet                            Подавите выходное напряжение
                --read-only                        Смонтируйте корневую файловую систему контейнера как доступную только для чтения
                --restart string                   Политика перезапуска, применяемая при выходе из контейнера (по умолчанию "нет")
                --rm                               Автоматически извлекать контейнер и связанные с ним
                                                    безымянные тома при выходе
                --runtime string                   Среда выполнения, используемая для этого контейнера
                --security-opt list                Параметры безопасности
                --shm-size bytes                   Размер /dev/shm
                --sig-proxy                        Прокси-сервер получал сигналы для процесса (по умолчанию true)
                --stop-signal string               Подайте сигнал на остановку контейнера
                --stop-timeout int                 Время ожидания (в секундах) для остановки контейнера
                --storage-opt list                 Опции драйвера хранилища для контейнера
                --sysctl map                       Параметры Sysctl (карта по умолчанию[])
                --tmpfs list                       Монтирование каталога tmpfs
            -t, --tty                              Выделите псевдо-TTY
                --ulimit ulimit                    Ограничить параметры (по умолчанию [])
            -u, --user string                      Имя пользователя или UID (format: [:])
                --userns string                    Используемое пространство имен пользователя
                --uts string                       Используемое пространство имен UTS
            -v, --volume list                      Привязать, смонтировать том
                --volume-driver string             Дополнительный драйвер тома для контейнера
                --volumes-from list                Монтируйте тома из указанных контейнеров
            -w, --workdir string                   Рабочий каталог внутри контейнера
