            Использование:  docker logs [OPTIONS] CONTAINER

            Извлеките логи из контейнера

            Псевдонимы:
            docker container logs, docker logs
            docker logs --stdout my_container
            Options:
                --details                    Показывать дополнительные сведения, предоставляемые журналам
            -f, --follow                     Следите за выводами журнала
            docker logs -f my_container
            docker logs -f --tail 50 my_container
                --since string               Показывать журналы с момента установки временной метки (например
                                            "2013-01-02T13:23:37Z") или относительный (например, "42 м"
                                            в течение 42 минут)
            docker logs --since "2023-07-20T15:00:00" my_container
            docker logs --since 1h my_container
            docker logs --since "2023-07-20T15:00:00" --until "2023-07-20T16:00:00" my_container
            -n, --tail string                Количество строк, отображаемых с конца журналов
            docker logs --tail 100 my_container
                                            (по умолчанию "all")
            -t, --timestamps                 Показывать временные метки
                --until string               Показывать журналы перед временной меткой (например,
                                            "2013-01-02T13:23:37Z") или относительный (например, "42 м"
                                            в течение 42 минут)
            docker logs --until "2023-07-20T16:00:00" my_container
            docker logs --until 10m my_container
