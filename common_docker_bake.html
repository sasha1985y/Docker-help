Build from a file

            Usage:  docker buildx bake [OPTIONS] [TARGET...]

            Build from a file

            Aliases:
            docker buildx bake, docker buildx f

            Options:
                --allow stringArray      Разрешить сборке доступ к указанным ресурсам
                --builder string         Переопределение настроенного экземпляра builder
                                        (по умолчанию "рабочий стол-linux")
                --call string            Установите метод оценки сборки ("проверка",
                                        "схема", "цели") (по умолчанию "построение")
                --check                  Сокращение от "--вызов=проверка" (по умолчанию)
            -D, --debug                  Включить ведение журнала отладки
            -f, --file stringArray       Build definition file
                --list string            Включить ведение журнала отладки
                --load                   Сокращение от "--set=*.output=type=docker"
                --metadata-file string   Запись метаданных результата сборки в файл
                --no-cache               Не используйте кэш при создании образа
                --print                  Распечатайте параметры, не создавая
                --progress string        Установите тип вывода хода выполнения ("автоматический",
                                        "quiet", "plain", "tty", "rawjson"). Воспользуйся
                                        простой для отображения выходных данных контейнера (по умолчанию "автоматически")
                --provenance string      Сокращение от "--set=*.attest=type=provenance"
                --pull                   Всегда пытайтесь извлечь все изображения, на которые есть ссылки
                --push                   Сокращение от "--set=*.output=type=registry"
                --sbom string            Сокращение от "--set=*.attest=type=sbom"
                --set stringArray        Переопределить целевое значение (например,
                                        "целевой шаблон.key=value")
            Экспериментальные команды и флаги скрыты. Set BUILDX_EXPERIMENTAL=1 to show them.
